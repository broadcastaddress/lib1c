function fill_registry(target,ext="ext",mimetype="type") export
    set=target.createRecordSet();
    for each pair in list() do
        row=set.add();
        row[ext]=pair.key;
        row[mimetype]=pair.value;
    enddo;
    set.write();
endfunction
